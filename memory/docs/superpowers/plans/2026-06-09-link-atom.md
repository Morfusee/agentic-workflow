```tsx
'use client'

import { cn } from '@/lib/utils'
import { useReducedMotion } from '@/hooks/useReducedMotion'
import { cva, type VariantProps } from 'class-variance-authority'
import { ArrowRight, ExternalLink } from 'lucide-react'
import NextLink from 'next/link'
import * as React from 'react'

const linkVariants = cva(
  [
    'group inline-flex items-center gap-1.5 font-semibold transition-colors duration-200',
    'focus-visible:outline-2 focus-visible:outline-offset-2 focus-visible:outline-brand-blue',
    '[&_svg]:pointer-events-none [&_svg]:shrink-0 [&_svg:not([class*=size-])]:size-4',
  ].join(' '),
  {
    variants: {
      variant: {
        inline:
          'text-brand-blue underline decoration-brand-blue/60 decoration-1 underline-offset-2 transition-[color,text-decoration-color,text-decoration-thickness,text-underline-offset] hover:text-brand-red hover:decoration-brand-red hover:decoration-2 hover:underline-offset-4',
        standalone: 'font-bold text-brand-blue hover:text-brand-red',
        'with-arrow': 'font-bold text-brand-blue hover:text-brand-red',
        quiet:
          'text-ink hover:text-brand-red hover:underline hover:decoration-brand-red/50 hover:underline-offset-4',
        inverse:
          'text-white underline decoration-white/60 decoration-1 underline-offset-2 transition-[text-decoration-color,text-decoration-thickness,text-underline-offset] hover:decoration-white hover:decoration-2 hover:underline-offset-4',
      },
      size: {
        sm: 'text-sm',
        md: 'text-base',
      },
      active: {
        true: 'text-brand-red decoration-brand-red',
      },
      disabled: {
        true: 'pointer-events-none cursor-not-allowed opacity-50',
      },
    },
    defaultVariants: {
      variant: 'inline',
      size: 'sm',
    },
  },
)

export type LinkVariant = NonNullable<VariantProps<typeof linkVariants>['variant']>
export type LinkSize = NonNullable<VariantProps<typeof linkVariants>['size']>

export interface LinkProps
  extends Omit<
      React.AnchorHTMLAttributes<HTMLAnchorElement>,
      'aria-label' | 'children' | 'href' | 'target' | 'rel'
    >,
    Omit<VariantProps<typeof linkVariants>, 'active' | 'disabled'> {
  href: string
  children?: React.ReactNode
  label?: string
  external?: boolean
  newTab?: boolean
  active?: boolean
  disabled?: boolean
  leadingIcon?: React.ReactNode
  trailingIcon?: React.ReactNode
  ariaLabel?: string
  'aria-label'?: string
  prefetch?: boolean
}

const hasProtocol = (href: string) => /^[a-z][a-z\d+.-]*:/i.test(href)

const isExternalHref = (href: string, external?: boolean) => {
  if (external !== undefined) return external
  if (!/^https?:\/\//i.test(href)) return false

  const baseUrl = process.env.NEXT_PUBLIC_BASE_URL
  if (!baseUrl) return true

  try {
    return new URL(href).origin !== new URL(baseUrl).origin
  } catch {
    return true
  }
}

export function Link({
  href,
  children,
  label,
  variant = 'inline',
  size = 'sm',
  external,
  newTab,
  active,
  disabled,
  leadingIcon,
  trailingIcon,
  ariaLabel,
  'aria-label': ariaLabelProp,
  className,
  prefetch,
  onClick,
  ...props
}: LinkProps) {
  const prefersReducedMotion = useReducedMotion()
  const content = children ?? label
  const isExternal = isExternalHref(href, external)
  const shouldUseAnchor = isExternal || hasProtocol(href) || href.startsWith('#')
  const shouldOpenInNewTab = newTab || isExternal
  const accessibleLabel = ariaLabel ?? ariaLabelProp
  const resolvedPrefetch = prefetch ?? process.env.NEXT_PUBLIC_BUILD_MODE !== 'static'
  const shouldShowDefaultIcon = (variant === 'standalone' || variant === 'with-arrow') && !trailingIcon

  const linkClassName = cn(linkVariants({ variant, size, active, disabled }), className)

  const handleClick: React.MouseEventHandler<HTMLAnchorElement> = (event) => {
    if (disabled) {
      event.preventDefault()
      return
    }

    onClick?.(event)
  }

  const innerContent = (
    <>
      {leadingIcon ? (
        <span aria-hidden="true" className="inline-flex shrink-0">
          {leadingIcon}
        </span>
      ) : null}
      {content ? <span>{content}</span> : null}
      {trailingIcon ? (
        <span aria-hidden="true" className="inline-flex shrink-0">
          {trailingIcon}
        </span>
      ) : null}
      {shouldShowDefaultIcon ? (
        <span
          aria-hidden="true"
          className={cn(
            'inline-flex shrink-0 transition-transform duration-200 group-hover:translate-x-0.5 group-focus-visible:translate-x-0.5',
            prefersReducedMotion && 'transition-none group-hover:translate-x-0 group-focus-visible:translate-x-0',
          )}
        >
          {isExternal ? <ExternalLink className="size-3.5" /> : <ArrowRight className="size-3.5" />}
        </span>
      ) : null}
    </>
  )

  if (disabled) {
    return (
      <a
        {...props}
        aria-disabled="true"
        aria-current={active ? 'page' : undefined}
        aria-label={accessibleLabel}
        className={linkClassName}
        onClick={handleClick}
      >
        {innerContent}
      </a>
    )
  }

  if (shouldUseAnchor) {
    return (
      <a
        {...props}
        href={href}
        target={shouldOpenInNewTab ? '_blank' : undefined}
        rel={shouldOpenInNewTab ? 'noopener noreferrer' : undefined}
        aria-current={active ? 'page' : undefined}
        aria-label={accessibleLabel}
        className={linkClassName}
        onClick={handleClick}
      >
        {innerContent}
      </a>
    )
  }

  return (
    <NextLink
      {...props}
      href={href}
      prefetch={resolvedPrefetch}
      aria-current={active ? 'page' : undefined}
      aria-label={accessibleLabel}
      className={linkClassName}
      onClick={handleClick}
    >
      {innerContent}
    </NextLink>
  )
}

export default Link
```
